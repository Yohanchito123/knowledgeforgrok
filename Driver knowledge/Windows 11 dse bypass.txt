security.h, option 1 to disable and 2 to enable.
Code:
#pragma once
 
#include "imports.h"
#include "memory.h"
 
#define in_range(x,a,b)    (x >= a && x <= b) 
#define get_bits( x )    (in_range((x&(~0x20)),'A','F') ? ((x&(~0x20)) - 'A' + 0xA) : (in_range(x,'0','9') ? x - '0' : 0))
#define get_byte( x )    (get_bits(x[0]) << 4 | get_bits(x[1]))
#define to_lower_c(Char) ((Char >= (char*)'A' && Char <= (char*)'Z') ? (Char + 32) : Char)
 
namespace security
{
    static UCHAR g_original_header[6] = { 0 };
    static UCHAR g_original_data[6] = { 0 };
    static bool  g_saved_originals = false;
    uintptr_t se_validate_image_header;
    uintptr_t se_validate_image_data;
 
    auto get_system_information(SYSTEM_INFORMATION_CLASS InformationClass) -> void* {
        unsigned long size = 32;
        char buffer[32];
 
        ZwQuerySystemInformation(InformationClass, buffer, size, &size);
 
        void* info = ExAllocatePoolZero(NonPagedPool, size, 'UD');
 
        if (!info)
            return nullptr;
 
        if (!NT_SUCCESS(ZwQuerySystemInformation(InformationClass, info, size, &size))) {
            ExFreePool(info);
            return nullptr;
        }
 
        return info;
    }
 
    auto get_kernel_module(const char* name) -> uintptr_t {
 
        const PRTL_PROCESS_MODULES info = (PRTL_PROCESS_MODULES)get_system_information(SystemModuleInformation);
 
        if (!info)
            return NULL;
 
        for (size_t i = 0; i < info->NumberOfModules; ++i) {
            const auto& mod = info->Modules[i];
 
            if (strcmp(to_lower_c((char*)mod.FullPathName + mod.OffsetToFileName), name) == 0) {
                const void* address = mod.ImageBase;
                ExFreePool(info);
                return (uintptr_t)address;
            }
        }
 
        ExFreePool(info);
        return NULL;
    }
 
    uintptr_t find_pattern(uintptr_t module_base, const char* pattern) {
        auto pattern_ = pattern;
        uintptr_t first_match = 0;
 
        if (!module_base) {
            return 0;
        }
 
        const auto nt = reinterpret_cast<IMAGE_NT_HEADERS*>(module_base + reinterpret_cast<IMAGE_DOS_HEADER*>(module_base)->e_lfanew);
 
        for (uintptr_t current = module_base; current < module_base + nt->OptionalHeader.SizeOfImage; current++) {
            if (!*pattern_) {
                return first_match;
            }
 
            if (*(BYTE*)pattern_ == '\?' || *(BYTE*)current == get_byte(pattern_)) {
                if (!first_match)
                    first_match = current;
 
                if (!pattern_[2])
                    return first_match;
 
                if (*(WORD*)pattern_ == '\?\?' || *(BYTE*)pattern_ != '\?')
                    pattern_ += 3;
 
                else
                    pattern_ += 2;
            }
            else {
                pattern_ = pattern;
                first_match = 0;
            }
        }
 
        return 0;
    }
 
    bool dse(int option) {
        
 
        if (option == 1)
        {
            auto ntoskrnl_base = get_kernel_module("ntoskrnl.exe");
 
            if (!ntoskrnl_base)
            {
                DbgPrintEx(0, 0, "Failed to get ntoskrnl.exe module base.\n");
                return STATUS_FAILED_DRIVER_ENTRY;
            }
 
            DbgPrintEx(0, 0, "ntoskrnl.exe base at: %p\n", ntoskrnl_base);
 
            se_validate_image_header = find_pattern(ntoskrnl_base,
                "48 8B C4 48 89 58 ? 57 48 81 EC A0 00 00 00 48 83 60"
            );
 
            se_validate_image_data = find_pattern(ntoskrnl_base,
                "48 83 EC 48 48 8B 05 ? ? ? ? 4C 8B D1"
            );
 
            if (!se_validate_image_header || !se_validate_image_data)
            {
                DbgPrintEx(0, 0, "Failed to find one or both function patterns.\n");
                return STATUS_NOT_FOUND;
            }
 
            const UINT8 patch[] = { 0xB8, 0x01, 0x00, 0x00, 0x00, 0xC3 };
 
            if (!g_saved_originals)
            {
                RtlCopyMemory(g_original_header, (void*)se_validate_image_header, sizeof(patch));
                RtlCopyMemory(g_original_data, (void*)se_validate_image_data, sizeof(patch));
                g_saved_originals = true;
            }
 
            if (!memory::write_to_wp((void*)se_validate_image_header, patch, sizeof(patch)))
            {
                DbgPrint("Failed to patch se_validate_image_header\n");
                return STATUS_ACCESS_VIOLATION;
            }
 
            if (!memory::write_to_wp((void*)se_validate_image_data, patch, sizeof(patch)))
            {
                DbgPrint("Failed to patch se_validate_image_data\n");
                return STATUS_ACCESS_VIOLATION;
            }
 
            DbgPrintEx(0, 0, "se_validate_image_header patched at %p\n", (void*)se_validate_image_header);
            DbgPrintEx(0, 0, "se_validate_image_data patched at %p\n", (void*)se_validate_image_data);
        }
        else if (option == 2)
        {
            if (!g_saved_originals)
            {
                DbgPrintEx(0, 0,
                    "Cannot restore because original bytes were never saved.\n"
                );
                return STATUS_INVALID_PARAMETER;
            }
 
            if (!memory::write_to_wp(
                (void*)se_validate_image_header,
                g_original_header,
                sizeof(g_original_header)))
            {
                DbgPrint("Failed to restore se_validate_image_header\n");
                return STATUS_ACCESS_VIOLATION;
            }
 
            if (!memory::write_to_wp(
                (void*)se_validate_image_data,
                g_original_data,
                sizeof(g_original_data)))
            {
                DbgPrint("Failed to restore se_validate_image_data\n");
                return STATUS_ACCESS_VIOLATION;
            }
 
            DbgPrintEx(0, 0, "se_validate_image_header restored at %p\n", (void*)se_validate_image_header);
            DbgPrintEx(0, 0, "se_validate_image_data restored at %p\n", (void*)se_validate_image_data);
        }
 
        return STATUS_SUCCESS;
    }
}
write_to_wp:
Code:
bool write_to_wp(void* dst, const void* src, size_t size)
{
KIRQL oldIrql = KeRaiseIrqlToDpcLevel();
 
unsigned __int64 cr0 = __readcr0();
 
__writecr0(cr0 & ~(1UL << 16));
 
RtlCopyMemory(dst, src, size);
 
__writecr0(cr0);
 
KeLowerIrql(oldIrql);
 
return true;
}

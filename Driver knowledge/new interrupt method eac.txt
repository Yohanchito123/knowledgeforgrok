Unhappy new interrupt method eac
I've been analyzing the latest devirtualized and cleaned EAC build following their recent update, and I noticed something new that wasn't in the previous version.

They're implementing a per-CPU DPC fan-out with a context blob for each core.
Based on what I'm seeing, this could be some form of stack profiling for presumably unbacked execution detection, though I can't say for certain.

This is definitely a new addition compared to the last build I analyzed.

Code:
__int64 sub_4150B0()
{
  ULONG i;
  ULONG j;
  ULONG ActiveProcessorCount;
  int v4;
  PRKDPC Dpc;
  _BYTE *P;
 
  ActiveProcessorCount = EAC::DecryptImport(KeQueryActiveProcessorCountEx)(0xFFFFu);
  v4 = sub_415290();
  if ( v4 >= 0 )
  {
    P = EAC::DecryptImport(ExAllocatePoolWithTag)(NonPagedPool, 768LL * ActiveProcessorCount, 0x62657377u);
    if ( P )
    {
      sub_4168E0(P, 0, 768LL * ActiveProcessorCount);
      for ( i = 0; i < ActiveProcessorCount; ++i )
      {
        _mm_lfence();
        Dpc = (PRKDPC)&P[768 * i];
        HIDWORD(Dpc[11].DpcData) = i;
        LOBYTE(Dpc[11].DpcData) = 0;
        EAC::DecryptImport(KeInitializeEvent)((PRKEVENT)&Dpc[1], NotificationEvent, 0);
        EAC::DecryptImport(KeInitializeDpc)(Dpc, DeferredRoutine, Dpc);
        EAC::DecryptImport(KeSetTargetProcessorDpc)(Dpc, i);
        EAC::DecryptImport(KeSetImportanceDpc)(Dpc, HighImportance);
        EAC::DecryptImport(KeInsertQueueDpc)(Dpc, 0, 0);
      }
      for ( j = 0; j < ActiveProcessorCount; ++j )
      {
        _mm_lfence();
        EAC::DecryptImport(KeWaitForSingleObject)(&P[768 * j + 64], Executive, 0, 0, 0);
        if ( P[768 * j + 760] )
        {
          _mm_lfence();
          sub_4149C0(&P[768 * j + 88]);
        }
      }
      EAC::DecryptImport(ExFreePoolWithTag)(P, 0x62657377u);
      return 0;
    }
    else
    {
      return 3221225626LL;
    }
  }
  else
  {
    _mm_lfence();
    return (unsigned int)v4;
  }
}
